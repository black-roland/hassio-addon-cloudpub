#include <tunables/global>

profile cloudpub flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  file,
  signal (send) set=(kill,term,int,hup,cont),
  network,

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Access to options.json and other files within your addon
  /data/** rw,

  # Read access to logs
  /root/.cache/cloudpub/client.log r,

  # Start new profile for service
  /usr/bin/clo cx -> clo,

  profile clo flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    # Receive signals from S6-Overlay
    signal (receive) peer=*_cloudpub,

    # Access to addon data and configuration
    /{data,config,addons}/** rw,

    # Allow clo to map its own binary into memory
    /usr/bin/clo rm,

    # Access required for service functionality
    /{bin,usr/bin}/{bash,echo} rix,
    /etc/{passwd,group,nsswitch.conf,hosts,resolv.conf,host.conf} r,
    /dev/tty rw,

    # Allow access to /root/.cache/
    /root/.cache/ rw,
    /root/.cache/** rwk,

    # Allow access to resources required by the Tokio runtime
    /sys/{,**} r,
    /proc/{,**} r,

    # Allow access to SSL certificates
    /etc/ssl/certs/** r,

    # Network access
    network inet stream,
    network inet6 stream,
  }
}
